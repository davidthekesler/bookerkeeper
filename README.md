CREATE TABLE "genre" (
  "id" serial primary key,
  "name" varchar(80)
);

CREATE TABLE "book" (
  "id" serial primary key,
  "title" varchar(120),
  "author" varchar(80),
  "image" varchar(1000),
  "date" DATE,
  "score" INT,
  "favorite" BOOLEAN,
  "genre_id" INT REFERENCES "genre"

);
